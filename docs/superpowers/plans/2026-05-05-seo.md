- [ ] **Step 3: Verify JSON-LD is valid JSON**

```bash
node -e "
const html = require('fs').readFileSync('index.html', 'utf8');
const match = html.match(/<script type=\"application\/ld\+json\">([\s\S]*?)<\/script>/);
if (!match) { console.error('No JSON-LD block found'); process.exit(1); }
try {
  const data = JSON.parse(match[1]);
  console.log('✓ Valid JSON-LD');
  console.log('  @graph entries:', data['@graph'].length);
  data['@graph'].forEach(e => console.log(' -', e['@type']));
} catch (e) {
  console.error('✗ Invalid JSON:', e.message);
  process.exit(1);
}
"
```
